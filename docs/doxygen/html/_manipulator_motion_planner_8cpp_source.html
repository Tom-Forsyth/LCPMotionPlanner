58 // Compute the joint displacement change to avoid collision at the next time step.
59 Eigen::VectorXd ManipulatorMotionPlanner::getCollisionDisplacementChange(const Eigen::VectorXd& displacementChange) const

75 }

80 size_t dim = contactPoints.size();

82 Eigen::MatrixXd M = Eigen::MatrixXd::Identity(dim, dim);
83
84 // Loop over each active rigid body.
85 int row = 0;
86 for (const auto& pair : contactPoints)
87 {
88 // Get distance, normal, and row contact jacobian.
89 double distance = pair.second.m_distance;
90 Eigen::Vector<double, 6> paddedNormal = Eigen::Vector<double, 6>::Zero();
91 paddedNormal.head(3) = pair.second.m_normal;
92 Eigen::MatrixXd contactJacobian = rigidBodies[pair.first].getContactJacobian();
93
94 // Form element of q.
95 q(row) = (distance - m_safetyDistance) + (((paddedNormal.transpose() * contactJacobian) * displacementChange));
96
97 // Loop over again to form entire row of M.
98 int col = 0;
99 for (const auto& pair2 : contactPoints)
100 {
101 Eigen::MatrixXd colContactJacobian = rigidBodies[pair2.first].getContactJacobian();
102 Eigen::VectorXd colPaddedNormal = Eigen::Vector<double, 6>::Zero();
103 colPaddedNormal.head(3) = pair2.second.m_normal;
104 M(row, col) = (((paddedNormal.transpose() * contactJacobian) * getNullSpaceTerm()) * colContactJacobian.completeOrthogonalDecomposition().pseudoInverse()) * colPaddedNormal;
105 col++;
106 }
107 row++;
108 }
109
110 // Solve LCP for compensating velocites.
111 LCPSolve::LCP solution = LCPSolve::LCPSolve(M, q);
112 Eigen::VectorXd compensatingVelocities = solution.z;
113
114 // Find the change in displacements based on compensating velocities.
115 Eigen::VectorXd collisionDisplacementChange = Eigen::VectorXd::Zero(m_dof);
116 int compVelIndex = 0;
117 int movableIndex = 0;
118 for (const RigidBody& body : rigidBodies)
119 {
120 // If the body is movable.
121 if (body.isMovable())
122 {
123 // If the body has an active contact.
124 if (contactPoints.count(movableIndex))
125 {
126 Eigen::MatrixXd contactJacobianInv = (body.getContactJacobian()).completeOrthogonalDecomposition().pseudoInverse();
127 Eigen::Vector<double, 6> paddedNormal = Eigen::Vector<double, 6>::Zero();
128 paddedNormal.head(3) = body.getContactPoint().m_normal;
129 collisionDisplacementChange += (contactJacobianInv * paddedNormal) * compensatingVelocities[compVelIndex];
130 compVelIndex++;
131 }
132 movableIndex++;
133 }
134 }
