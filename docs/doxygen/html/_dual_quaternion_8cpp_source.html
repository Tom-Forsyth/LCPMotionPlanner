77 // Power.
78 DualQuaternion DualQuaternion::pow(const double& exponent) const
79 {
80 // Get axis and angle from the real portion of the dual quaternion.
81 Eigen::AngleAxis<double> axisAngle(m_real);
82 Eigen::Vector3d axis = axisAngle.axis();
83 double angle = axisAngle.angle();
84
85 // Get the translation vector from the dual portion of the dual quaternion.
86 Eigen::Vector4d transVecHomo = 2 * (m_dual * m_real.conjugate()).coeffs();
87 Eigen::Vector3d transVec = transVecHomo.head(3);
88
89 // Compute the ttranslation along the screw axis by projecting the displacement vector, and the moment vector.
90 double displacement = 0;
91 Eigen::Vector3d moment = Eigen::Vector3d::Zero();
92 if (angle < 1e-10) // Should this be mod pi?
93 {
94 axis = transVec.normalized(); // Axis has no meaning at singularities so we choose this.
95 displacement = transVec.dot(axis);
96 }
97 else
98 {
99 displacement = transVec.dot(axis);
100 moment = 0.5 * ((transVec.cross(axis)) + (transVec - displacement * axis) * (1 / tan(angle / 2)));
101 }
102
103 // Construct new dual quaternion.
104 double const1 = exponent * angle / 2;
105 double const2 = exponent * displacement / 2;
106 DualNumber dualNum1{ cos(const1), -const2 * sin(const1) };
107 DualNumber dualNum2{ sin(const1), const2 * cos(const1) };
108
109 std::vector<DualNumber> dualVec;
110 dualVec.reserve(3);
111 for (size_t i = 0; i < 3; i++) {
112 DualNumber dualNum3(axis(i), moment(i));
113 dualVec.emplace_back(dualNum3 * dualNum2);
114 }
115
116 Eigen::Quaterniond realQuat(dualNum1.m_real, dualVec[0].m_real, dualVec[1].m_real, dualVec[2].m_real);
117 Eigen::Quaterniond dualQuat(dualNum1.m_dual, dualVec[0].m_dual, dualVec[1].m_dual, dualVec[2].m_dual);
118 return DualQuaternion(realQuat, dualQuat);
119 }
