59 {
60 // Keep running product of displacements rather than storing each.

62

116 // Update spatial jacobian for each body.
117 void RigidBodyChain::updateSpatialJacobians()
118 {
119 // Loop over each body, adding columns for movable joints and updating the member.
120 Eigen::MatrixXd spatialJacobian = Eigen::MatrixXd::Zero(6, m_nMovableBodies);
121 Eigen::Matrix4d displacementProduct = Eigen::Matrix4d::Identity();
122 int bodyIndex = 0;
123 for (RigidBody& body : m_rigidBodies)
124 {
125 if (body.isMovable())
126 {
127 // Compute column of jacobian.
128 Eigen::Vector<double, 6> twistCoord = body.getJointTwistCoord();
129 if (bodyIndex == 0)
130 {
131 spatialJacobian.col(bodyIndex) = twistCoord;
132 }
133 else
134 {
135 spatialJacobian.col(bodyIndex) = Kinematics::adjoint(displacementProduct) * twistCoord;
136 }
137
138 // Compute next iteration's product of i-1.
139 displacementProduct *= body.getRelativeTransformation();
140 bodyIndex++;
141 }
142
143 // Set the body's spatial jacobian with zero padding.
144 body.setSpatialJacobian(spatialJacobian);
145 }
